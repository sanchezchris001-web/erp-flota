<script>
const socket = io();
const ROL = "{{user.rol}}";

// KPI
const cd=document.getElementById("cd");
const co=document.getElementById("co");
const ud=document.getElementById("ud");
const uo=document.getElementById("uo");
const ui=document.getElementById("ui");

// listas
const asignaciones=document.getElementById("asignaciones");
const conductores=document.getElementById("conductores");
const unidades=document.getElementById("unidades");
const movimientos=document.getElementById("movimientos");

// selects
const conductorSelect=document.getElementById("conductorSelect");
const unidadSelect=document.getElementById("unidadSelect");
const conductorFinal=document.getElementById("conductorFinal");
const unidadFinal=document.getElementById("unidadFinal");
const unidadInh=document.getElementById("unidadInh");
const unidadHab=document.getElementById("unidadHab");

// inputs
const obsInh=document.getElementById("obsInh");
const nombre=document.getElementById("nombre");
const placa=document.getElementById("placa");

// usuarios
const mu_user=document.getElementById("mu_user");
const mu_pass=document.getElementById("mu_pass");
const mu_rol=document.getElementById("mu_rol");

const modal=document.getElementById("modalUsuarios");
const listaUsuarios=document.getElementById("listaUsuarios");

socket.on("actualizar",()=>cargar());

function cargar(){
fetch("/datos").then(r=>r.json()).then(d=>{

cd.innerText=d.stats.conductores_disponibles;
co.innerText=d.stats.conductores_ocupados;
ud.innerText=d.stats.unidades_disponibles;
uo.innerText=d.stats.unidades_ocupadas;
ui.innerText=d.stats.unidades_inhabilitadas;

// asignaciones
asignaciones.innerHTML =
d.asignaciones.map(a=>`<li>${a.conductor} → ${a.unidad}</li>`).join("");

// conductores
conductores.innerHTML=d.conductores.map(c=>`
<li>${c.nombre} - ${c.estado}
<button onclick="editarConductor(${c.id})">✏️</button>
<button onclick="eliminarConductor(${c.id})">🗑️</button>
</li>`).join("");

// unidades
unidades.innerHTML=d.unidades.map(u=>`
<li>${u.placa} - ${u.estado}
<button onclick="editarUnidad(${u.id})">✏️</button>
<button onclick="eliminarUnidad(${u.id})">🗑️</button>
</li>`).join("");

// selects
conductorSelect.innerHTML=d.conductores.map(c=>`<option value="${c.id}">${c.nombre}</option>`).join("");
unidadSelect.innerHTML=d.unidades.map(u=>`<option value="${u.id}">${u.placa}</option>`).join("");

conductorFinal.innerHTML=conductorSelect.innerHTML;
unidadFinal.innerHTML=unidadSelect.innerHTML;
unidadInh.innerHTML=unidadSelect.innerHTML;

unidadHab.innerHTML=d.unidades.filter(u=>u.estado==="inhabilitado")
.map(u=>`<option value="${u.id}">${u.placa}</option>`).join("");

});

// historial
if(ROL==="admin"){
fetch("/movimientos").then(r=>r.json()).then(d=>{
movimientos.innerHTML =
d.map(m=>`<li>${m.tipo} - ${m.obs || ""}</li>`).join("");
});
}
}

// ================= CRUD =================

function crearConductor(){
fetch("/crear_conductor",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({nombre:nombre.value})}).then(()=>cargar());
}

function editarConductor(id){
let nuevo=prompt("Nuevo nombre:");
if(!nuevo) return;
fetch("/editar_conductor",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({id,nombre:nuevo})}).then(()=>cargar());
}

function eliminarConductor(id){
if(confirm("Eliminar?")){
fetch("/eliminar_conductor",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({id})}).then(()=>cargar());
}
}

// UNIDADES
function crearUnidad(){
fetch("/crear_unidad",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({placa:placa.value})}).then(()=>cargar());
}

function editarUnidad(id){
let nueva=prompt("Nueva placa:");
if(!nueva) return;
fetch("/editar_unidad",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({id,placa:nueva})}).then(()=>cargar());
}

function eliminarUnidad(id){
if(confirm("Eliminar?")){
fetch("/eliminar_unidad",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({id})}).then(()=>cargar());
}
}

// ================= OPERACIONES =================

function asignar(){
fetch("/asignar",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({conductor_id:conductorSelect.value,unidad_id:unidadSelect.value})})
.then(()=>cargar());
}

function finalizar(){
fetch("/finalizar",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({conductor_id:conductorFinal.value,unidad_id:unidadFinal.value})})
.then(()=>cargar());
}

function inhabilitar(){
fetch("/inhabilitar",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({unidad_id:unidadInh.value,observacion:obsInh.value})})
.then(()=>cargar());
}

function habilitar(){
fetch("/habilitar",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({unidad_id:unidadHab.value})})
.then(()=>cargar());
}

// ================= USUARIOS =================

function abrirModal(){modal.style.display="flex";cargarUsuarios();}
function cerrarModal(){modal.style.display="none";}

function cargarUsuarios(){
fetch("/usuarios").then(r=>r.json()).then(d=>{
listaUsuarios.innerHTML=d.map(u=>`
<li>${u.username} (${u.rol})
<button onclick="editarUsuario(${u.id})">✏️</button>
<button onclick="eliminarUsuario(${u.id})">🗑️</button>
</li>`).join("");
});
}

function crearUsuario(){
fetch("/crear_usuario",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({username:mu_user.value,password:mu_pass.value,rol:mu_rol.value})})
.then(()=>cargarUsuarios());
}

function editarUsuario(id){
let username=prompt("Usuario:");
let password=prompt("Password:");
let rol=prompt("Rol:");
fetch("/editar_usuario",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({id,username,password,rol})})
.then(()=>cargarUsuarios());
}

function eliminarUsuario(id){
if(confirm("Eliminar?")){
fetch("/eliminar_usuario",{method:"POST",headers:{"Content-Type":"application/json"},
body:JSON.stringify({id})})
.then(()=>cargarUsuarios());
}
}

// INIT
cargar();
